//DEFINING VARIABLES
int luminosity; //level of light
float imthirsty = 0; //level of thirsty [0,100] lower is better
int thresholdlight = 150; //detect night and day
int thresholdtemp = 20; //detect mean temperature 
float imhungry = 0; //level of hungry [0,100] lower is better
int foodbutton; //give food, decrease hungry
int waterbutton; //give water, decrease thirsty
float life = 100; //status of life
float degree; //temperature in Celsius degree
DHT dht(temp, DHTTYPE); //temperature sensor
Adafruit_SH1106G display = Adafruit_SH1106G(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET); //OLED display

unsigned long time_hungry = 0;
unsigned long time_thirsty = 0;
unsigned long time_food = 0;
unsigned long time_water = 0;
unsigned long time_blink_hungry = 0;
unsigned long time_blink_thirsty = 0;
unsigned long time_blink = 0;
unsigned long time_print = 0;

int redledstate = LOW;
int greenledstate = LOW;

//SETUP PIN
void setup() {

  pinMode(temp, INPUT);
  pinMode(light, INPUT);
  pinMode(food, INPUT);
  
  pinMode(feed, OUTPUT);
  pinMode(cold, OUTPUT);
  pinMode(awake, OUTPUT);
  Serial.begin(9600);

  dht.begin();
  display.begin(i2c_Address, true);
  display.clearDisplay();
  display.drawBitmap(0, 0, image_data_HELLOarray, 128, 64, 1); // Display hello
  display.display();  
  delay(3000);

}

void loop() {

luminosity = analogRead(light);
foodbutton = digitalRead(food); //feeding by pressing red button
waterbutton = digitalRead(water); //give water by pressing blue button
degree = dht.readTemperature();

if (luminosity > thresholdlight){
  digitalWrite(awake, HIGH); //pet is always awake when yellow led is on  //awake when the light is on

  //BLINKING EYES
  if (millis() - time_blink > 1000){
    display.clearDisplay();
    display.drawBitmap(0, 0, image_data_CLOSEDarray, 128, 64, 1);
    display.display();
    display.clearDisplay();
    time_blink = millis();}
  else if (millis() - time_blink > 200){
    display.clearDisplay();
    display.drawBitmap(0, 0, image_data_OPENarray, 128, 64, 1);
    display.display();
    display.clearDisplay();}

  //HUNGRY STATE
  if (foodbutton == LOW){ //food diminuish only when pet is awake
    if (millis() >= time_food + 1000){ //food diminuish every second
      imhungry = imhungry + 0.2;
      time_food = millis();}}
      
    //STARVING
    if (imhungry > 70){ //red led blink only when pet is starving

      //PLAIN EYES    
      if (millis() - time_blink_hungry > 1000){
        display.clearDisplay();
        display.drawBitmap(0, 0, image_data_CRYarray, 128, 64, 1);
        display.display();
        display.clearDisplay();}
      
      //LIFE DECREASING
      life = life - 0.2;

      //RED LED BLINKING
      if (millis() - time_hungry > 1000){
        if (redledstate == LOW) {
          redledstate = HIGH;}   
        else {
            redledstate = LOW;}

      digitalWrite(feed, redledstate);
      time_hungry = millis();}}

    
  //GIVE FOOD
  if (foodbutton == HIGH) //you can feed the pet only when awake
    imhungry--; 
    
  //THIRSTY STATE
  if ((waterbutton == LOW) && (degree > thresholdtemp)) //water diminuish only when pet is awake and temperature is greater than 24°
    if (millis() > time_water + 1000){
      imthirsty = imthirsty + 0.2;
      time_water = millis();}
    
    //DEAD BECAUSE TOO THIRSTY
    if (imthirsty > 70){

      //PLAIN EYES
      if (millis() - time_blink_thirsty > 1000){
        display.clearDisplay();
        display.drawBitmap(0, 0, image_data_CRYarray, 128, 64, 1);
        display.display();
        display.clearDisplay();}

      //LIFE DECREASING
      life = life - 0.2; 

      //GREEN LED BLINKING IF THIRSTY >70%
      if (millis() - time_thirsty > 1000){
        if (greenledstate == LOW) {
        greenledstate = HIGH;} 
      else {
        greenledstate = LOW;}

      digitalWrite(cold, greenledstate);
      time_thirsty = millis();}}
  
  //GIVE WATER
  if (waterbutton == HIGH) 
    imthirsty--; 


    }


//CORRECTING VARIABLES' RANGE TO MAKE PET DIE
if (imhungry < 0)
  imhungry = 0;

if (imhungry > 100){
  imhungry = 100;
  life=0;
  }

if (imthirsty < 0)
  imthirsty = 0;

if (imthirsty > 100) {
  imthirsty = 100;
  life = 0;
  }
    
if (life < 0)
  life = 0;

if (luminosity < thresholdlight){ //pet is sleeping, yellow led is off

  //YELLOW LED IS OFF
  digitalWrite(awake, LOW);

  //CLOSED EYES
  display.clearDisplay();
  display.drawBitmap(0, 0, image_data_CLOSEDarray, 128, 64, 1); // Display closed eyes
  display.display();}



//DISPLAYING STATUS ON SERIAL
if (millis() - time_print > 1000){
  time_print = millis();
  Serial.print("Status|"); Serial.print("Degree |"); Serial.print("Hungry  |"); Serial.print("Thirsty  |"); Serial.println("Life     |");
  if (life == 0)
    Serial.print("DIE :(");
  else if ((luminosity < thresholdlight))
    Serial.print("Sleep");
  else if (luminosity > thresholdlight)
    Serial.print("Awake");    
  Serial.print(" | "); Serial.print(degree); Serial.print(" | "); Serial.print(imhungry); Serial.print("%"); Serial.print("  | "); Serial.print(imthirsty); Serial.print("%"); Serial.print("  | "); Serial.print(life); Serial.print("%"); Serial.println(" | ");
  }
}
